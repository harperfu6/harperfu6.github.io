from matplotlib.colors import ListedColormap

def plot_decision_boundary(clf, X, y, axes=[-1.5, 2.5, -1, 1.5], alpha=0.5, contour=True):
    x1s = np.linspace(axes[0], axes[1], 100)
    x2s = np.linspace(axes[2], axes[3], 100)
    x1, x2 = np.meshgrid(x1s, x2s)
    X_new = np.c_[x1.ravel(), x2.ravel()]
    y_pred = clf.predict(X_new).reshape(x1.shape)
    custom_cmap = ListedColormap(['#fafab0','#9898ff','#a0faa0'])
    plt.contourf(x1, x2, y_pred, alpha=0.3, cmap=custom_cmap)
    if contour:
        custom_cmap2 = ListedColormap(['#7d7d58','#4c4c7f','#507d50'])
        plt.contour(x1, x2, y_pred, cmap=custom_cmap2, alpha=0.8)
    plt.plot(X[:, 0][y==0], X[:, 1][y==0], "yo", alpha=alpha)
    plt.plot(X[:, 0][y==1], X[:, 1][y==1], "bs", alpha=alpha)
    plt.axis(axes)
    plt.xlabel(r"$x_1$", fontsize=18)
    plt.ylabel(r"$x_2$", fontsize=18, rotation=0)

AdaBoost (Adaptive Boosting)¶

予測器を複数直列に並べ，前の予測器が過少適合したインスタンスに少しだけ余分に注意を払う．

具体的には，分類に失敗したインスタンスの相対的な重みを上げ，更新された重みを使って次の分類器で訓練する．これをひたすら繰り返す．
この逐次的な学習テクニックは勾配降下法に似ているが，コスト関数を最小化するために一つの予測器のパラメータを操作するのではなく，アンサンブルに予測器を追加する形でアンサンブルを改良していく．

逐次的な学習テクニックは，前の予測器の訓練/評価が終わらなければ次の予想器の訓練ができないため，並列化できないという大きな欠点を抱えている． 従って，バギングやペースティングと比べてスケーラビリティが低い．

scikit-learnにはSAMMEというマルチクラスバージョンのアダブーストを使っている．
クラスが２つだけなら，SAMMEとアダブーストは同じである．
また，予測器が確率を推計できる場合には，SAMME.Rという変種を使える．これは，クラスに属する確率を使うもので，一般にSAMMEよりも性能が高い．

次のコードは，決定株（max_depth=1の決定木，つまり一つの判断ノードと２つの葉ノードから構成される決定木)を使ったAdaBoostaClassifilerクラス（決定株はデフォルトのベース推定器である）

Gradient Boosting¶

勾配ブースティングも，アダブーストと同様にして前の予測器を改良した予測器を逐次的に加えていく方法．
違いは，前の予測器の残差(residual)に適合させようとする点．

ベースとして決定木を使った勾配ブースティングを勾配ブースティング木や勾配ブースティング決定木（GBRT:gradient boosted regression tree）と呼ぶ

def plot_predictions(regressors, X, y, axes, label=None, style="r-", data_style="b.", data_label=None):
    x1 = np.linspace(axes[0], axes[1], 500)
    y_pred = sum(regressor.predict(x1.reshape(-1, 1)) for regressor in regressors)
    plt.plot(X[:, 0], y, data_style, label=data_label)
    plt.plot(x1, y_pred, style, linewidth=2, label=label)
    if label or data_label:
        plt.legend(loc="upper center", fontsize=16)
    plt.axis(axes)

import numpy as np
from sklearn.model_selection import train_test_split
from sklearn.metrics import mean_squared_error

X_train, X_val, y_train, y_val = train_test_split(X, y, random_state=49)

gbrt = GradientBoostingRegressor(max_depth=2, n_estimators=120, random_state=42)
gbrt.fit(X_train, y_train)

errors = [mean_squared_error(y_val, y_pred)
          for y_pred in gbrt.staged_predict(X_val)]
bst_n_estimators = np.argmin(errors)

gbrt_best = GradientBoostingRegressor(max_depth=2,n_estimators=bst_n_estimators, random_state=42)

実装
warm_start=Trueとすることで，すでにfit済みのモデルに追加で訓練できる（漸進的に訓練できる）ので，はじめにn_estimatorsを決めなくても良い．

In [29]:
gbrt = GradientBoostingRegressor(max_depth=2, warm_start=True, random_state=42)

min_val_error = float('inf')
error_going_up = 0
for n_estimators in range(1, 120):
    gbrt.n_estimators = n_estimators
    gbrt.fit(X_train, y_train)
    y_pred = gbrt.predict(X_val)
    val_error = mean_squared_error(y_val, y_pred)
    if val_error < min_val_error:
        min_val_error = val_error
        error_going_up = 0
    # もし最小値が連続して５回上がったらストップをかける
    else:
        error_going_up +=1
        if error_going_up == 5:
            break # early stopping

stochastic gradient boosting¶

また，GradientBoostingRegressorクラスは，個々の決定木を訓練するために使われるインスタンスの割合を指定するsubsampleハイパーパラメータをサポートしている．これにより，無作為に選んだインスタンスで学習できるためここの分類器のバイアスを上げた分，アンサンブルのバリアンスを下げることができるしかも，全てのインスタンスを使うわけではないので訓練スピードも上がる
このテクニックを確率的勾配ブースティングという

スタッキング¶

複数の予測器の集計において，ハード投票のようなつまらない関数を使うのではなく，それらを特徴量としてさらに訓練し最終的な予測値を算出すること．
この最後の予測器のことをブレンダー，メタ学習器と呼ぶ

XGBoost¶

特徴：

  • 勾配ブースティングとランダムフォレストを組み合わせたアンサンブル学習
  • sparseなデータについて，予め分岐の方向を決めるアルゴリズムを導入
  • アウトオブコア学習が可能
  • 複数のパラメータ調整によるチューニングが必要，最適化はグリッドサーチや交差検証を複数行う
  • 汎化能力をあげるために，学習率(eta)を下げていき，その都度最適化を行う
